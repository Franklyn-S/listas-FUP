SUB-ROTINA inverter(array, posicao)
	SE posicao < len(array)/2 ENTAO
	INICIO
		tmp = array[posicao]
		nova_posicao = len(array)-posicao-1
		array[posicao] = array[nova_posicao]
		array[nova_posicao] = tmp
		inverter(array, posicao+1)
	FIM
FIM SUB-ROTINA inverter
